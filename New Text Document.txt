#include <bits/stdc++.h>
using namespace std;
int ara[99][99];
int main()
{int tc,x;cin>>tc;while(tc--){cin>>x;for(int i=0; i<x; i++){for(int j=1; j<i; j++){cin>>ara[i][j];}}for(int i=x-2; i>=0; i--){for(int j=i; j>=0; j--){ara[i][j]=max((ara[i+1][j]+ara[i][j]),(ara[i+1][j+1]+ara[i][j]));}}cout<<ara[0][0]<<endl;}
}
